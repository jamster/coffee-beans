Select the SVG Element

svg = d3.select('svg')

Append a circle

circle = svg.append('circle')

Give the circle a position and radius

circle
  .attr('cx', 400)
  .attr('cy', 100)
  .attr('r', 40)

Transition to a new position, radius, and fill color

circle
  .transition()
  .duration(800)
  .attr('cx', 500)
  .attr('cy', 150)
  .attr('fill', 'orange')
  .attr('r', 200)

Create some data

some_data = [1,2,3,4,5,4,3,2,1]

Bind the data to the circles

circles = svg
  .selectAll('circle')
  .data(some_data)
  .enter().append('circle')
