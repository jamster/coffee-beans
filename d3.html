Tutorial

Try copying these code blocks into the console to the right, and run them.

Select the SVG Element

svg = d3.select('#chart')

Append a circle

circle = svg.append('circle')

Give the circle a position and radius

circle
  .attr('cx', 100)
  .attr('cy', 100)
  .attr('r', 40)

Transition to a new position, radius, and fill color

circle
  .transition()
  .duration(800)
  .attr('cx', 200)
  .attr('cy', 200)
  .attr('fill', 'orange')
  .attr('r', 200)

Create some data

some_data = [1,2,3,4,5,4,3,2,1]

Bind the data to the circles

svg.selectAll('circle')
  .data(some_data)
  .enter().append('circle')
  
circles = svg.selectAll('circle')

Make a simple visualization

circles
  .transition()
  .duration(800)
  .attr('stroke', '#666')
  .attr('fill', '#444')
  .attr('cx', (d,i) -> 40+40*i)
  .attr('cy', (d,i) -> 40+40*i)
  .attr('r', (d,i) -> 4*d)

Transition representation

circles
  .transition()
  .duration(800)
  .attr('fill', (d) ->
    d3.hsl(40*d,0.6,0.7/d)
  )
  .attr('cx', 210)
  .attr('stroke', '#fff')
